  			if (nota > 10) {
  				Alerts.showAlert("Nota acima de 10", "Cuidado", "A nota máxima deve ser 10", AlertType.WARNING);

  								"insert into feedback (id_entrega, nota, comentario, id_aluno) values (?, ?, ?, ?)");

  						st2.setFloat(2, nota);
  						st2.setString(3, comentario);
  						st2.setInt(4, aluno.getId_aluno());
  						st2.executeUpdate();
  						Alerts.showAlert("Sucesso", "Feedback dado!!!", "O feedback foi armazenado com sucesso",

                  parseCSV(data) {

                      const headers = rows[0].split(';');

                                  obj[headers[j].trim()] = values[j].trim();

                              result.push(obj);

                      console.log("Resultado do parseCSV:", result);
